CREATE VIEW VIEW_FIN_TOTAL_RECEBIMENTO_DIA(
	ID_CONTA_CAIXA,
    DATA_RECEBIMENTO,
    TOTAL)
AS
select
  PR.ID_CONTA_CAIXA, PR.DATA_RECEBIMENTO, SUM(PR.VALOR_RECEBIDO) AS TOTAL
from
  FIN_PARCELA_RECEBIMENTO PR

inner join FIN_TIPO_RECEBIMENTO tp on
	PR.ID_FIN_TIPO_RECEBIMENTO = tp.ID
inner join EMPRESA emp on
	tp.ID_EMPRESA = emp.ID_EMPRESA
group by PR.ID_CONTA_CAIXA, PR.DATA_RECEBIMENTO;
